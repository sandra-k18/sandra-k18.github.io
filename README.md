// ─── SURVEY FLOW ─────────────────────────────────────────────────────────────

function Survey({ onDone }) {
  const [step, setStep] = useState(0);
  const [answers, setAnswers] = useState({});
  const [tgNick, setTgNick] = useState("");
  const [hovered, setHovered] = useState(null);

  function save(key, val) {
    const updated = { ...answers, [key]: val };
    setAnswers(updated);
    return updated;
  }

  function mockSaveToSheets(data) {
    console.log("📊 [Mock Google Sheets] Saving:", data);
  }

  function Btn({ id, children, onClick, variant }) {
    return (
      <button
        style={{ ...S.btn(variant || "primary"), ...(hovered === id ? S.btnHovered : {}) }}
        onMouseEnter={() => setHovered(id)}
        onMouseLeave={() => setHovered(null)}
        onClick={onClick}
      >
        {children}
      </button>
    );
  }

  // Step 0: going to Bali?
  if (step === 0) return (
    <div style={S.actionsWrap}>
      <div style={S.sectionLabel}>Вопрос 1 из 5</div>
      <Btn id="yes" onClick={() => { save("going", "да"); setStep(10); }}>✈️ Да</Btn>
      <Btn id="no" variant="secondary" onClick={() => { save("going", "нет"); setStep(1); }}>Нет</Btn>
    </div>
  );

  // Step 1: invited to test?
  if (step === 1) return (
    <div style={S.actionsWrap}>
      <div style={S.sectionLabel}>Тебя пригласили протестировать функционал чат-бота?</div>
      <Btn id="inv-yes" onClick={() => { const d = save("invited", "да"); mockSaveToSheets(d); onDone(); }}>Да</Btn>
      <Btn id="inv-no" variant="secondary" onClick={() => { save("invited", "нет"); setStep(2); }}>Нет</Btn>
    </div>
  );

  // Step 2: how did you get here?
  if (step === 2) {
    const [otherDest, setOtherDest] = useState("");
    const [otherExp, setOtherExp] = useState("");
    return (
      <div style={S.actionsWrap}>
        <div style={S.sectionLabel}>Расскажи, как ты попал в чат-бот?</div>
        <Btn id="curious" onClick={() => { const d = save("source", "Из любознательности"); mockSaveToSheets(d); onDone(); }}>Из любознательности</Btn>
        <div style={{ display: "flex", flexDirection: "column", gap: 6 }}>
          <Btn id="other-dest" onClick={() => { const d = save("source", `Искал помощника по путешествию, но не на Бали. Куда: ${otherDest}`); mockSaveToSheets(d); onDone(); }}>
            Искал помощника по планированию путешествия, но не на Бали
          </Btn>
          <input
            style={{ ...S.input, fontSize: 13, padding: "8px 12px" }}
            placeholder="Куда ты хотел(а) отправиться?"
            value={otherDest}
            onChange={e => setOtherDest(e.target.value)}
          />
        </div>
        <div style={{ display: "flex", flexDirection: "column", gap: 6 }}>
          <Btn id="wrong" variant="secondary" onClick={() => { const d = save("source", `Думал(а), что это про другое. Ожидания: ${otherExp}`); mockSaveToSheets(d); onDone(); }}>
            Думал(а), что это вообще про другое
          </Btn>
          <input
            style={{ ...S.input, fontSize: 13, padding: "8px 12px" }}
            placeholder="Какие были ожидания от чат-бота?"
            value={otherExp}
            onChange={e => setOtherExp(e.target.value)}
          />
        </div>
      </div>
    );
  }

  // Step 10: first time?
  if (step === 10) return (
    <div style={S.actionsWrap}>
      <div style={S.sectionLabel}>Вопрос 2 из 5 · Впервые?</div>
      <Btn id="first-yes" onClick={() => { save("firstTime", "да"); setStep(11); }}>Да, впервые</Btn>
      <Btn id="first-no" variant="secondary" onClick={() => { save("firstTime", "нет"); setStep(11); }}>Нет, был(а) раньше</Btn>
    </div>
  );

  // Step 11: purpose
  if (step === 11) return (
    <div style={S.actionsWrap}>
      <div style={S.sectionLabel}>Вопрос 3 из 5 · Цель поездки</div>
      {["Путешествие", "Путешествие + удалённая работа, зимовка", "Релокация"].map((opt, i) => (
        <Btn key={i} id={`goal-${i}`} onClick={() => { save("goal", opt); setStep(12); }}>{opt}</Btn>
      ))}
    </div>
  );

  // Step 12: duration
  if (step === 12) return (
    <div style={S.actionsWrap}>
      <div style={S.sectionLabel}>Вопрос 4 из 5 · Как долго?</div>
      {["1–2 недели", "3–4 недели", "1–1,5 месяца", "Посмотрю по ситуации, пока сроков нет"].map((opt, i) => (
        <Btn key={i} id={`dur-${i}`} variant={i === 3 ? "secondary" : "primary"} onClick={() => { save("duration", opt); setStep(13); }}>{opt}</Btn>
      ))}
    </div>
  );

  // Step 13: with whom
  if (step === 13) return (
    <div style={S.actionsWrap}>
      <div style={S.sectionLabel}>Вопрос 5 из 5 · С кем едешь?</div>
      {["Один/одна", "С парой (парень/девушка, муж/жена)", "С семьёй с детьми", "С компанией 2+ друзей"].map((opt, i) => (
        <Btn key={i} id={`with-${i}`} onClick={() => { save("with", opt); setStep(14); }}>{opt}</Btn>
      ))}
    </div>
  );

  // Step 14: Telegram
  if (step === 14) return (
    <div style={S.actionsWrap}>
      <div style={S.sectionLabel}>Необязательно</div>
      <div style={{ fontSize: 13.5, color: "#c0b090", marginBottom: 4, fontFamily: "Arial, sans-serif" }}>
        Если ты готов(а) принять участие в интервью о том, как ты обычно планируешь путешествия, оставь свой ник в Telegram. Саша свяжется в течение 2 рабочих дней.
      </div>
      <div style={S.inputRow}>
        <input
          style={S.input}
          placeholder="@username"
          value={tgNick}
          onChange={e => setTgNick(e.target.value)}
        />
      </div>
      <Btn id="survey-done" onClick={() => {
        const d = save("telegram", tgNick || "не указан");
        mockSaveToSheets(d);
        onDone();
      }}>
        {tgNick ? "Отправить и продолжить →" : "Пропустить и продолжить →"}
      </Btn>
    </div>
  );

  return null;
}
